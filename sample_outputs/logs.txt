def reverse_in_parentheses(s: str) -> str:
    """
    Reverses the characters enclosed within each pair of parentheses in a given string,
    starting from the innermost one, and returns the resulting string with all parentheses removed.

    Args:
        s (str): The input string containing lowercase letters and parentheses.

    Returns:
        str: The resulting string with all parentheses removed and characters reversed within parentheses.
    """
    while '(' in s:
        # Find the innermost pair of parentheses
        start = s.rfind('(')
        end = s.find(')', start)
        
        # Reverse the substring within the innermost parentheses
        s = s[:start] + s[start + 1:end][::-1] + s[end + 1:]
    
    return s
